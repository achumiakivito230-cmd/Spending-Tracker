Create UIModule factory (similar pattern to DataModule from Task 1) that manages menu state with a single delegated click-outside handler.

**Insert UIModule after DataModule (around line 1600+):**

```javascript
const UIModule = (() => {
  let menuStack = [];  // Stack of open menus: ['budget'] or ['budget', 'label-actions']
  let clickOutsideListener = null;

  function attachClickOutside() {
    if (clickOutsideListener) return;  // Already attached, avoid duplicate listener

    clickOutsideListener = function(e) {
      // Check if click is outside ALL open menus
      if (menuStack.length === 0) {
        document.removeEventListener('click', clickOutsideListener);
        clickOutsideListener = null;
        return;
      }

      const topMenu = menuStack[menuStack.length - 1];
      let isOutside = true;

      // Build list of open menu elements to check
      if (topMenu === 'budget') {
        const menuEl = document.getElementById('budget-input-menu');  // ID or selector for budget menu
        if (menuEl && menuEl.contains(e.target)) isOutside = false;
      } else if (topMenu === 'label-actions') {
        const menuEl = document.getElementById('label-actions-menu');  // ID or selector for label menu
        if (menuEl && menuEl.contains(e.target)) isOutside = false;
      }

      if (isOutside) {
        closeMenu();  // Close top menu
      }
    };

    document.addEventListener('click', clickOutsideListener);
  }

  function closeMenu() {
    menuStack.pop();
    if (menuStack.length === 0 && clickOutsideListener) {
      document.removeEventListener('click', clickOutsideListener);
      clickOutsideListener = null;
    }
  }

  return {
    openMenu(menuName) {
      menuStack.push(menuName);
      attachClickOutside();
    },
    closeMenu() {
      closeMenu();
    },
    isMenuOpen(menuName) {
      return menuStack.includes(menuName);
    },
    getMenuStack() {
      return [...menuStack];
    }
  };
})();
```

**Key principles:**
- Single `clickOutsideListener` function, not recreated for each menu
- `attachClickOutside()` checks if listener already exists before attaching
- Menu stack tracks open menus in order
- `closeMenu()` pops from stack and removes listener when stack is empty
- No per-menu anonymous listeners — all handled through single delegated handler
