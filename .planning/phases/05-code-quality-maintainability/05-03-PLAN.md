<objective>
Implement chart rendering memoization (QUAL-04) to prevent jank on screen transitions.

Purpose: Chart SVG is expensive to build (data filtering, group aggregation, DOM element creation). Currently, switching from history screen to chart screen rebuilds SVG even if data hasn't changed. Memoization caches rendered SVG and only rebuilds when data changes or filter changes.

Output: ChartModule factory with:
- Encapsulated cache state (cachedHTML, cachedChartState with hash/period/labelFilter)
- computeDataHash() for deterministic state comparison
- render() method that checks cache before rebuilding
- Event listener on DataModule.data-changed to invalidate cache on data mutations
- Integration with existing filter buttons (period, label) to invalidate cache on filter change
</objective>

const ChartModule = (() => {
  // Private memoization cache
  let cachedChartState = null;  // {hash, period, labelFilter}
  let cachedHTML = null;

  // Compute deterministic hash of current chart inputs
  function computeDataHash(data, period, labelFilter) {
    // Create a string representation of current chart state
    const str = JSON.stringify({
      txCount: data.length,
      dataHash: data.map(tx => `${tx.id}-${tx.amount}`).join('|'),
      period,
      labelFilter
    });
    // For simple comparison, return the string itself
    // (In production, could use crypto.subtle.digest for true hash)
    return str;
  }

  // Validate that chart state is still the same
  function isCacheValid(currentData, period, labelFilter) {
    if (!cachedChartState || !cachedHTML) return false;

    const currentHash = computeDataHash(currentData, period, labelFilter);
    return cachedChartState.hash === currentHash;
  }

  // Invalidate cache when data changes
  function invalidateCache() {
    cachedChartState = null;
    cachedHTML = null;
  }

  // Build chart HTML (copied from existing renderChart logic at lines 2850-2862)
  function buildChartHTML(data, period, labelFilter) {
    // [Copy existing renderChart implementation here]
    // Return the SVG/HTML string

    // Filter data based on period and labelFilter (existing logic)
    // Group and aggregate by label
    // Build SVG using existing SVG building code
    // Return the HTML

    // NOTE: This is where lines 2850-2862 logic goes
    // Keep all the existing chart building logic unchanged
    // Just wrap it in memoization
  }

  // Listen to data changes from DataModule
  if (DataModule && DataModule.addEventListener) {
    DataModule.addEventListener('data-changed', () => {
      invalidateCache();
    });
  }

  // Public API
  return {
    render(data, period, labelFilter) {
      // Check if we can reuse cached output
      if (isCacheValid(data, period, labelFilter) && cachedHTML) {
        return cachedHTML;
      }

      // Build new chart
      const html = buildChartHTML(data, period, labelFilter);

      // Cache the result
      const hash = computeDataHash(data, period, labelFilter);
      cachedChartState = { hash, period, labelFilter };
      cachedHTML = html;

      return html;
    },

    invalidateCache() {
      invalidateCache();
    }
  };
})();
