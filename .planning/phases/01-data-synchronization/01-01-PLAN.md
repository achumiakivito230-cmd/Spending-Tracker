  <action>
    Open https://supabase.com/dashboard/project/exzyruzxfshexpdlkaeq/sql/new and run the following SQL exactly as written. This migration is idempotent (safe to re-run).

    ```sql
    -- Step 1: Add updated_at to expenses
    alter table expenses
      add column if not exists updated_at timestamptz default now();

    -- Step 2: Create trigger function (idempotent via CREATE OR REPLACE)
    create or replace function update_updated_at()
    returns trigger language plpgsql as $$
    begin
      new.updated_at = now();
      return new;
    end;
    $$;

    -- Step 3: Attach trigger to expenses
    drop trigger if exists expenses_updated_at on expenses;
    create trigger expenses_updated_at
      before update on expenses
      for each row execute function update_updated_at();

    -- Step 4: Add updated_at to labels
    alter table labels
      add column if not exists updated_at timestamptz default now();

    -- Step 5: Attach trigger to labels
    drop trigger if exists labels_updated_at on labels;
    create trigger labels_updated_at
      before update on labels
      for each row execute function update_updated_at();

    -- Step 6: Enable Realtime publication for both tables
    alter publication supabase_realtime add table expenses;
    alter publication supabase_realtime add table labels;
    ```
