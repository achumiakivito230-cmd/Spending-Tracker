```javascript
async function loadHistory() {
  if (_historyLoading) return _historyLoading;
  _historyLoading = _doLoadHistory();
  try { await _historyLoading; } finally { _historyLoading = null; }
}

async function _doLoadHistory() {
  const targetScreen = screen;
  const list = document.getElementById('tx-list');
  list.innerHTML = `<div style="color:var(--muted);font-size:14px;padding:28px 0;text-align:center">Loading…</div>`;
  document.getElementById('filter-row').innerHTML = '';
  activeFilter = null;

  if (!db) { list.innerHTML = `<div style="color:var(--muted);font-size:14px;padding:28px 0;text-align:center">No connection.</div>`; return; }
  const { data, error } = await db.from('expenses').select('*').order('created_at', { ascending: false }).limit(200);

  if (error || !data) {
    logError('loadHistory', error);
    list.innerHTML = `<div style="color:var(--muted);font-size:14px;padding:28px 0;text-align:center">Could not load transactions — tap to retry</div>`;
    toast('Could not load transactions — check your connection');
    return;
  }

  if (!Array.isArray(data)) return;
  if (screen !== targetScreen) return;
  allTxData = data;
  renderFilterChips(data);
  renderHistory(data, null);
}
```

Key points:
- `const targetScreen = screen` is captured BEFORE the await — this is the critical line for STATE-04.
- `if (screen !== targetScreen) return` sits after the await and before all DOM writes and state assignment.
- The `if (!Array.isArray(data)) return` guard from Plan 01 is preserved in this new _doLoadHistory body — do not add it a second time.
- The DOM write (`list.innerHTML = Loading…`) and `activeFilter = null` remain in _doLoadHistory (before the await) so the loading state still shows on explicit navigation.
